## Requirements

### Create `packages/core/src/store/ScopedRegistry.ts`

```typescript
interface AtomEntry<Value = unknown> {
  atom: Atom<Value>;
  state: AtomState<Value>;
  metadata: AtomMetadata;
}

class ScopedRegistry {
  // Storage
  private entries: Map<symbol, AtomEntry>;
  private nameToId: Map<string, symbol>;
  private store: Store;

  constructor(store: Store);

  // Single entry point for atom registration
  ensure<Value>(atom: Atom<Value>, stateFactory: () => AtomState<Value>): AtomEntry<Value>;

  // Lookup
  get<Value>(id: symbol): AtomEntry<Value> | undefined;
  getByName<Value>(name: string): AtomEntry<Value> | undefined;
  has(id: symbol): boolean;

  // Bulk operations
  getAll(): Map<symbol, AtomEntry>;
  getAllIds(): symbol[];

  // Serialization
  getStateAsRecord(): Record<string, unknown>;
  getStateAsMap(): Map<string, unknown>;

  // Metadata
  getName(atom: { id: symbol }): string;
  getMetadata(id: symbol): AtomMetadata | undefined;

  // Cleanup
  clear(): void;
  size(): number;
}
```

### Create `packages/core/src/store/types.ts` (or extend existing)

```typescript
export interface AtomMetadata {
  name: string;
  createdAt: number;
  type: 'primitive' | 'computed' | 'writable';
}

export interface AtomState<Value = unknown> {
  value: Value;
  subscribers: Set<(value: Value) => void>;
  dependents: Set<Atom<any>>;
}

export interface AtomEntry<Value = unknown> {
  atom: Atom<Value>;
  state: AtomState<Value>;
  metadata: AtomMetadata;
}
```

## Implementation Rules

1. **NO imports from `atom-registry.ts`** — fully self-contained
2. **Type inference** — `ensure()` should infer Value type from atom
3. **Lazy metadata** — generate name if `atom.name` is undefined (`atom-${size + 1}`)
4. **Duplicate handling** — if atom already registered, return existing entry (no-op)
5. **Fast path** — `has()` check before any Map operations
6. **No side effects** — no console.log, no global state mutation
