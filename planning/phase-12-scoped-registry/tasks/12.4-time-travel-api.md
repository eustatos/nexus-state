### 1. Define Snapshot type

```typescript
// packages/core/src/types.ts

/**
 * Immutable snapshot of store state at a point in time
 */
export interface StoreSnapshot {
  /** Serialized state: { atomName: value } */
  state: Record<string, unknown>;
  /** Timestamp when snapshot was taken */
  timestamp: number;
  /** Optional label for debugging */
  label?: string;
}

/**
 * Time-travel manager for store
 */
export interface TimeTravel {
  /** Take a snapshot of current state */
  snapshot(label?: string): StoreSnapshot;

  /** Restore state from snapshot */
  restore(snapshot: StoreSnapshot, options?: { silent?: boolean }): void;

  /** Go back one step */
  back(): StoreSnapshot | null;

  /** Go forward one step */
  forward(): StoreSnapshot | null;

  /** Get current position in history */
  currentIndex: number;

  /** Total history length */
  historyLength: number;

  /** Clear history */
  clear(): void;
}
```

### 2. Implement TimeTravelManager

```typescript
// packages/core/src/utils/time-travel.ts

import type { Store, StoreSnapshot, TimeTravel } from '../types';

export class TimeTravelManager implements TimeTravel {
  private history: StoreSnapshot[] = [];
  private index = -1;
  private store: Store;
  private maxSize: number;

  constructor(store: Store, maxSize = 100) {
    this.store = store;
    this.maxSize = maxSize;
  }

  snapshot(label?: string): StoreSnapshot {
    const snap: StoreSnapshot = {
      state: this.store.getState(),
      timestamp: Date.now(),
      label,
    };

    // Truncate future if we went back
    if (this.index < this.history.length - 1) {
      this.history = this.history.slice(0, this.index + 1);
    }

    this.history.push(snap);
    this.index = this.history.length - 1;

    // Enforce max size
    if (this.history.length > this.maxSize) {
      this.history.shift();
      this.index--;
    }

    return snap;
  }

  restore(snapshot: StoreSnapshot, options?: { silent?: boolean }): void {
    const { silent } = options || {};

    Object.entries(snapshot.state).forEach(([key, value]) => {
      // Find atom by name in store's registry
      // This requires store to expose setByName() or we iterate
      const atom = this.findAtomByName(key);
      if (atom) {
        this.store.set(atom, value, { silent, timeTravel: true });
      }
    });
  }

  back(): StoreSnapshot | null {
    if (this.index <= 0) return null;
    this.index--;
    return this.history[this.index];
  }

  forward(): StoreSnapshot | null {
    if (this.index >= this.history.length - 1) return null;
    this.index++;
    return this.history[this.index];
  }

  get currentIndex(): number {
    return this.index;
  }

  get historyLength(): number {
    return this.history.length;
  }

  clear(): void {
    this.history = [];
    this.index = -1;
  }

  private findAtomByName(name: string): any {
    // Iterate through store's registry to find atom by name

}
```

### 3. Add methods to StoreImpl

```typescript
// packages/core/src/store/StoreImpl.ts

class StoreImpl implements Store {
  private timeTravel: TimeTravelManager | null = null;

  // Called by constructor or enableTimeTravel()
  private initTimeTravel(maxSnapshots = 100): void {
    this.timeTravel = new TimeTravelManager(this as unknown as Store, maxSnapshots);
  }

  /**
   * Enable time-travel debugging
   * @param maxSnapshots Maximum snapshots to keep in memory
   */
  enableTimeTravel(maxSnapshots = 100): void {
    if (this.timeTravel) return; // already enabled
    this.initTimeTravel(maxSnapshots);
    this.snapshot(); // Take initial snapshot
  }

  /**
   * Get time-travel API (only if enabled)
   */
  getTimeTravel(): TimeTravel | null {
    return this.timeTravel;
  }

  /**
   * Take snapshot of current state
   */
  snapshot(label?: string): StoreSnapshot | null {
    return this.timeTravel?.snapshot(label) ?? null;
  }

  /**
   * Restore from snapshot
   */
  restore(snapshot: StoreSnapshot, options?: { silent?: boolean }): void {
    this.timeTravel?.restore(snapshot, options);
  }
}
```
