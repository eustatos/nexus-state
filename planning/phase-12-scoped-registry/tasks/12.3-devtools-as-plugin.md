# Task 12.3: Convert DevTools to Optional Plugin

| Meta | Value |
|------|-------|
| **Task** | Convert built-in DevTools to optional plugin (tree-shakeable) |
| **Priority** | P1 — enables bundle size reduction |
| **Estimated effort** | Medium |
| **Breaking changes** | No — backward compatible via re-exports |
| **Dependencies** | Task 12.2 |

## Context

Currently `DevToolsIntegration` is tightly coupled to `StoreImpl`. DevTools code is always in the bundle even when not used.

After this task: DevTools is an **optional plugin** — zero overhead if not imported.

**Files to modify:**
- `packages/core/src/store/DevToolsIntegration.ts` → becomes plugin
- `packages/core/src/store/StoreImpl.ts` → remove hardcoded DevTools
- `packages/core/src/index.ts` → add devtools export

## Architecture

### Before:

```
StoreImpl
  └─ devTools: DevToolsIntegration  // always present
       ├─ trackStateChange()
       ├─ trackAction()
       ├─ serializeState()
       └─ createInterceptedGetter/Setter()
```

### After:

```
StoreImpl
  └─ emits events: 'atom:registered', 'state:changed', 'action:tracked'

DevToolsPlugin (optional, in separate file)
  ├─ subscribes to store events
  ├─ connects to Redux DevTools
  └─ exported via @nexus-state/core/debug entry point
```

## Implementation

### 1. Create `packages/core/src/plugins/devtools.ts`

```typescript
import type { Store, Plugin, Atom } from '../types';

interface DevToolsOptions {
  name?: string;
  enabled?: boolean;
  maxHistory?: number;  // for time-travel
}

export class DevToolsPlugin implements Plugin {
  private connection: any;
  private history: Array<{ state: Record<string, unknown>; action: string }> = [];
  private store: Store | null = null;
  private options: Required<DevToolsOptions>;

  constructor(options?: DevToolsOptions) {
    this.options = {
      name: 'NexusState',
      enabled: options?.enabled ?? process.env.NODE_ENV !== 'production',
      maxHistory: options?.maxHistory ?? 50,
    };
  }

  // Plugin interface
  apply(store: Store): void {
    if (!this.options.enabled) return;
    this.store = store;
    this.connect();
  }

  private connect(): void {
    if (typeof window === 'undefined') return;
    const ext = (window as any).__REDUX_DEVTOOLS_EXTENSION__;
    if (!ext) return;

    this.connection = ext.connect({ name: this.options.name });

    // Listen for time-travel from DevTools UI
    this.connection.subscribe((message: any) => {
      if (message.type === 'DISPATCH' && message.payload.type === 'JUMP_TO_STATE') {
        this.handleTimeTravel(message.payload.index);
      }
    });
  }

  // Called by store on state changes
  trackStateChange(atom: Atom<any>, value: unknown): void {
    if (!this.connection || !this.store) return;

    const state = this.store.getState();
    this.connection.send(
      { type: 'SET', atom: atom.name || atom.id.toString() },
      state
    );

    this.history.push({ state, action: `set:${atom.name || atom.id.toString()}` });
    if (this.history.length > this.options.maxHistory) {
      this.history.shift();
    }
  }

  // Called by store on actions
  trackAction(actionName: string, payload: unknown): void {
    if (!this.connection || !this.store) return;

    this.connection.send(
      { type: actionName, payload },
      this.store.getState()
    );

    this.history.push({ state: this.store.getState(), action: actionName });
    if (this.history.length > this.options.maxHistory) {
      this.history.shift();
    }
  }

  private handleTimeTravel(index: number): void {
    if (!this.store) return;
    const snapshot = this.history[index];
    if (!snapshot) return;

    // Restore state — store will handle with { timeTravel: true }
    Object.entries(snapshot.state).forEach(([key, value]) => {
      // Find atom by name and set value
      // This requires store to expose setByName() or similar
    });
  }

  // Public API for manual cleanup
  disconnect(): void {
    this.connection?.disconnect();
    this.connection = null;
  }
}

/**
 * Factory function for convenience
 */
export function devtools(options?: DevToolsOptions): Plugin {
  return new DevToolsPlugin(options);
}
```

### 3. Add event system to Store (minimal)

```typescript
type StoreEventHandler = (...args: unknown[]) => void;

class StoreImpl {
  private handlers = new Map<string, StoreEventHandler[]>();

  on(event: string, handler: StoreEventHandler): () => void {
    const list = this.handlers.get(event) || [];
    list.push(handler);
    this.handlers.set(event, list);
    return () => {
      const idx = list.indexOf(handler);
      if (idx >= 0) list.splice(idx, 1);
    };
  }

  private emit(event: string, ...args: unknown[]): void {
    const list = this.handlers.get(event) || [];
    list.forEach(h => {
      try { h(...args); } catch { /* ignore plugin errors */ }
    });
  }
}
```
